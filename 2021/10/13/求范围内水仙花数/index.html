求水仙花数的简单封装

// 求 minNum 到 maxNum 之间（左开右闭）所有的水仙花数
// 水仙花数：abc == a³ + b³ + c³
function getFlower(minNum, maxNum) {
    var result = [];
    for(var i = minNum; i < maxNum; i++) {
        var abc = '' + i;
        var sum = 0;
        for(var j = 0; j < abc.length; j++) {
            sum += Math.pow(abc[j], 3)
        }
        if (sum == i) {
            result.push(i)
        }
    }
    return result
}
console.log(getFlower(100,999))
